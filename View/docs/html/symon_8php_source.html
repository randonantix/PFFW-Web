117 function SetConf($lanif, $wanif)
118 {
120 $ncpu= 1;
121 if (($hwncpu= $this->_getSysCtl('hw.ncpu')) !== FALSE) {
122 if (preg_match('|^hw.ncpu=(\d+)$|ms', $hwncpu, $match)) {
123 $ncpu= $match[1];
124 }
125 }
126
127 $cpus= '';
128 for ($i= 0; $i < $ncpu; $i++) {
129 $cpus.= " cpu($i),\n";
130 }
131
132 $others= " mem,\n pf,\n mbuf,\n";
133
134 $ifs= " if(lo0),\n if($lanif),\n if($wanif),\n";
135
136 $partitions= $this->_getPartitions();
137 $parts= '';
138 $disks= array();
139 foreach ($partitions as $part => $mdir) {
140 if (preg_match('|/dev/((\w+\d+)[a-z]+)|', $part, $match)) {
141 $parts.= ' df('.$match[1]."),\n";
142 if (!in_array($match[2], $disks)) {
143 $disks[]= $match[2];
144 }
145 }
146 }
147
148 $ios= '';
149 foreach ($disks as $io) {
150 $ios.= " io($io),\n";
151 }
152
153 $sensors= '';
154 if (($sensorslist= $this->GetSensors()) !== FALSE) {
155 if (isset($sensorslist['temp']) && count($sensorslist['temp']) > 0) {
156 foreach ($sensorslist['temp'] as $s) {
157 $sensors.= " sensor($s),\n";
158 }
159 }
160
161 if (isset($sensorslist['fan']) && count($sensorslist['fan']) > 0) {
162 foreach ($sensorslist['fan'] as $s) {
163 $sensors.= " sensor($s),\n";
164 }
165 }
166 }
167
168 $proclist= array(
169 'httpd',
170 'sshd',
171 'dnsmasq',
172 'dhcpd',
173 'ftp-proxy',
174 );
175 $procs= '';
176 foreach ($proclist as $p) {
177 $procs.= " proc($p),\n";
178 }
179
180 $conf= "\n".$cpus.$others.$ifs.$ios.$parts.$sensors.$procs;
181
182 $re= '|(\s*monitor\s*\{\h*)([^\}]*)(\s*\})|ms';
183 $retval= $this->ReplaceRegexp('/etc/symon.conf', $re, '${1}'.$conf.'${3}');
184
185 $conf= preg_replace('/( )/ms', ' ', $conf);
186
187 $re= '|(\s*source\s*127\.0\.0\.1\s*\{\s*accept\s*\{\h*)([^\}]*)(\h*\})|ms';
188 $retval&= $this->ReplaceRegexp('/etc/symux.conf', $re, '${1}'.$conf.' ${3}');
189
190 return $retval;
191 }

201 function RenderLayout($layout, $width= 700, $heigth= 250)
202 {
204 global $SRC_ROOT, $symon, $cache, $session, $chr2html, $runtime;
205
206 require_once ("$SRC_ROOT/View/symon/class_session.inc");
207 require_once ("$SRC_ROOT/View/symon/class_layout.inc");
208
209 $session->getform('start');
210 $session->getform('end');
211 $session->getform('width', $width);
212 $session->getform('heigth', $heigth);
213 $session->getform('layout');
214 $session->getform('timespan');
215 $session->getform('size', 'custom');
216
217 $l = new Layout($layout);
218
219 $graphs = $l->render(false);
220
221 require_once("$SRC_ROOT/View/symon/class_rrdtool.inc");
222
223 $graphFiles= array();
224 foreach ($graphs as $title => $g) {
225 if (preg_match("/^([0-9a-f]+)/", $g, $match)) {
226 $key = $match[1];
227 $filename = $cache->getfilename($key);
228 $extension = get_extension($filename);
229
230 if ($extension == 'txt') {
231 $definition = load($filename);
232 $cache->expire_key($key);
233 $rrdtool = new RRDTool();
234 $graph_file = $cache->obtain_filecache($key);
235 $result = $rrdtool->graph($graph_file, $definition);
236 } else {
237 $graph_file = $filename;
238 $result = 1;
239 }
240 $graphFiles[$title]= basename($graph_file);
241 }
242 }
243
244 return Output(json_encode($graphFiles));
245 }
