32/*
33 * Session class
34 * - provides support for user interaction
35 * - unifies all access to information for the user: the cache used for this
36 * session is allocated here.
37 * - parses information obtained from forms, holds defaults for this
38 * information if none was supplied by the user
39 *
40 * Currently this class does not persist user supplied information across
41 * multiple requests.
42 */

57 function getform($name, $val= '') {
58 global $symon, $View, $Submenu;
59
60 if (isset($_SESSION)) {
61 if ($Submenu) {

63 }
64 else {
65 $session= &$_SESSION[$View->Model]['Graphs'];
66 }
67 }
68
69 if (isset($_REQUEST[$name])) {
70 $value = $_REQUEST[$name];
71 // POSTed values have precedence over session vars
72 $session[$name]= $value;
73 }
74
75 if (isset($session[$name])) {
76 $value= $session[$name];
77 }
78
79 if ($val != '') {
80 $value= $val;
81 }
82
83 if (!isset($value)) {
84 $value = $this->getformdefault($name);
85 } else {
86 $value = $this->getformconstrained($name, $value);
87 }
88
89 // Do not forget to save the new/default value to session var
90 $session[$name]= $value;
91
92 /* unroll items if named var selected */
93 if ($symon['defaults'][$name]['type'] == 'named') {
94 $s = $symon['defaults'][$name]['namedvalues'][$value];
95 if (is_array($s)) {
96 foreach ($s as $k => $v) {
97 $this->vars->set($k, $v);
98 }
99 }
100 }
101 $this->vars->set($name, $value);
102
103 return $value;
104 }

123 function getformdefault($name) {
124 global $symon;
125
126 if (!isset($symon['defaults'][$name]['default'])) {
127 runtime_error('session: unknown variable "'.$name.'" requested');

135 function getformconstrained($name, $value) {
136 global $symon;
137
138 if (!isset($symon['defaults'][$name])) {
139 runtime_error('session: unknown variable "'.$name.'" requested');
140 }
141
142 switch ($symon['defaults'][$name]['type']) {
143 case 'named':
144 if (!isset($symon['defaults'][$name]['namedvalues'][$value])) {
145 $value = $this->getformdefault($name);
146 }
147 break;
148 case 'integer':
149 if ($value < $symon['defaults'][$name]['bounds']['min'] ||
150 $value > $symon['defaults'][$name]['bounds']['max']) {
151 $value = $this->getformdefault($name);
152 }
153 break;
154 default:
155 runtime_error('session: unknown variable type "'.$symon['defaults'][$name]['type'].'" requested');
156 break;
157 }
158
159 return $value;
160 }
