468 function _getIfConfig($if)
469 {
470 $file= $this->confDir."hostname.".$if;
471 if (file_exists($file)) {
472 if (($contents= $this->GetFile($file)) !== FALSE) {
473 $contents= explode("\n", $contents);
474
475 $inet= array('', '', '', '', '', '', '', '', '');
476 $nwid= array('', '', '');
477 $hostap= array('');
478 $wifi= array('');
479 $up= array('');
480
481 foreach ($contents as $line) {
482 if (preg_match("/^(inet|dhcp|autoconf)\s*(\S*)\s*(\S*)\s*(\S*)\s*(\S*)$/", $line, $match)) {
483 $inet= $match;
484 // OpenBSD 6.6 onwards uses hex if mask in hostname.if file
485 // Convert if mask from hex to ip: 0xffffff00 -> 255.255.255.0
486 if ($inet[3] !== '') {
487 $inet[3]= long2ip(hexdec($inet[3]));
488 }
489
490 $output= $this->RunShellCommand("/sbin/ifconfig $if | /usr/bin/grep -E '(inet |lladdr )'");
491 // lladdr aa:bb:cc:dd:ee:ff
492 // inet 192.168.0.1 netmask 0xffffff00 broadcast 192.168.0.255
493 if (preg_match('/lladdr\s*(\S+)/m', $output, $match)) {
494 $inet[]= $match[1];
495 } else {
496 $inet[]= '';
497 }
498 if (preg_match('/inet\s*(\S+)/m', $output, $match)) {
499 $inet[]= $match[1];
500 } else {
501 $inet[]= '';
502 }
503 if (preg_match('/netmask\s*(\S+)/m', $output, $match)) {
504 $inet[]= long2ip(hexdec($match[1]));
505 } else {
506 $inet[]= '';
507 }
508 if (preg_match('/broadcast\s*(\S+)/m', $output, $match)) {
509 $inet[]= $match[1];
510 } else {
511 $inet[]= '';
512 }
513 }
514 // Search in hostname.$if contents
515 // inet 192.168.0.1 0xffffff00
516 // nwid mynwid wpakey mykey
517 // mediaopt hostap
518 else if (preg_match('/(nwid|join)\s+(\S+)\s*(\S*)\s*(\S*)/', $line, $match)) {
519 $nwid= array($match[2], $match[3], $match[4]);
520 }
521 else if (preg_match('/mediaopt (hostap)/', $line, $match)) {
522 $hostap= array($match[1]);
523 }
524 }
525
526 if ($this->isWifiIf($if)) {
527 $wifi= array('wifi');
528 }
529
530 if ($this->isIfUp($if)) {
531 $up= array('up');
532 }
533 return json_encode(array_merge(array_slice($inet, 1), $nwid, $hostap, $wifi, $up));
534 }
535 }
536 return FALSE;
537 }

755 function SetIf($if, $type, $ip, $mask, $bc, $opt, $nwid, $keytype, $key, $hostap)
756 {
757 global $Re_Ip;
758
759 // OpenBSD 6.6 uses hex if mask in hostname.if file
760 // Convert if mask from ip to hex: 255.255.255.0 -> 0xffffff00
761 if ($mask !== '') {
762 $mask= '0x'.dechex(ip2long($mask));
763 }
764
765 // PFFW supports only these configuration options
766 // Trim whitespace caused by empty strings
767 $ifconf= trim("$type $ip $mask $bc $opt");
768
769 if ($nwid !== '') {
770 $ifconf.= "\n".trim("nwid $nwid");
771 if ($keytype !== '' && $key !== '') {
772 $ifconf.= ' '.trim("$keytype $key");
773 }
774 }
775
776 if ($hostap == 'on') {
777 $ifconf.= "\nmediaopt hostap";
778 }
779
780 // @todo Args passed to this function are validated by the Controller, but we can implement some semantic validation too
781 if (preg_match("/^inet\s*$Re_Ip\s*[x0-9a-f]+\s*($Re_Ip|).*$/m", $ifconf)
782 || preg_match('/^(dhcp|autoconf)\s*NONE\s*NONE\s*NONE.*$/m', $ifconf)
783 || preg_match('/^(dhcp|autoconf)$/m', $ifconf)
784 || preg_match('/(dhcp|autoconf)/m', $ifconf) && ($hostap === '')) {
787 return file_put_contents($this->confDir.'hostname.'.$if, $ifconf.PHP_EOL);
788 }
789 else {
790 Error(_('Unsupported interface configuration').": $ifconf");
791 }
792 return FALSE;
793 }

1051 function NetStart($if= '')
1052 {
1053 $cmd= array();
1054
1055 $intif= $this->_getIntIf();
1056 if ($intif !== FALSE) {
1057 $intif= trim($intif, '"');
1058 if ($if == '' || $if == $intif) {
1059 if ($this->isWifiIf($intif)) {
1060 // Clear all wifi configuration, ifconfig down and /etc/netstart don't clear them
1061 $cmd[]= "/sbin/ifconfig $intif -mediaopt hostap -nwid -wpakey -nwkey 2>&1";
1062 }
1063 $cmd[]= "/sbin/ifconfig $intif down 2>&1";
1064 }
1065 } else {
1066 Error('Cannot get intif');
1067 }
1068
1069 // Down extif too, in case the gateway changes
1070 $extif= $this->_getExtIf();
1071 if ($extif !== FALSE) {
1072 $extif= trim($extif, '"');
1073 if ($if == '' || $if == $extif) {
1074 if ($this->isWifiIf($extif)) {
1075 // Clear all wifi configuration, ifconfig down and /etc/netstart don't clear them
1076 $cmd[]= "/sbin/ifconfig $extif -mediaopt hostap -nwid -wpakey -nwkey 2>&1";
1077 }
1078 $cmd[]= "/sbin/ifconfig $extif down 2>&1";
1079 }
1080 } else {
1081 Error('Cannot get extif');
1082 }
1083
1084 if ($if == '' || $if == $extif) {
1085 // Delete the arp entry for the old gateway, in case the new gateway has the same IP address
1086 // Otherwise, the system cannot reach the new gateway, continues to use the old ethernet address,
1087 // ping, nslookup, or others don't work (extif becomes effectively down).
1088 $gateway= $this->getSystemGateway();
1089 if ($gateway !== '') {
1090 $cmd[]= "/usr/sbin/arp -nd $gateway 2>&1";
1091 }
1092 // Delete the current default route, so we are sure we load the pf rules after the new default route is set
1093 $cmd[]= "/sbin/route delete default 2>&1";
1094 }
1095
1096 // Pass $if as arg, passing empty string is fine
1097 $cmd[]= "/bin/sh /etc/netstart $if 2>&1";
1098
1099 // /etc/netstart does not up (autoconf?) ifs on OpenBSD 7.0 anymore
1100 if ($intif !== FALSE) {
1101 if ($if == '' || $if == $intif) {
1102 $cmd[]= "/sbin/ifconfig $intif up 2>&1";
1103 }
1104 }
1105
1106 if ($extif !== FALSE) {
1107 if ($if == '' || $if == $extif) {
1108 $cmd[]= "/sbin/ifconfig $extif up 2>&1";
1109 }
1110 }
1111
1112 // Implode with ';' not '&&', any of the commands can fail, but we should exec the others
1113 $cmdline= implode(' ; ', $cmd);
1114
1115 exec($cmdline, $output, $retval);
1116 $errout= implode("\n", $output);
1117 Error($errout);
1118
1119 // Refresh the pf rules too
1120 $retval&= $this->reloadPfRules();
1121
1122 if ($retval === 0) {
1123 return TRUE;
1124 }
1125 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, "Netstart failed: $errout");
1126 return FALSE;
1127 }
