279function IsDateTime($datetime)
280{
281 global $MonthDays;
282
284 if (preg_match('/^\d{10,}$/', $datetime)) {
285 $datetime= str_split($datetime, 2);
287 //$Year= $datetime[0] + 0;
288 $month= $datetime[1] + 0;
289 $day= $datetime[2] + 0;
290 $hour= $datetime[3] + 0;
291 $min= $datetime[4] + 0;
292 if (($month <= 12)
293 && ($day <= $MonthDays[$month])
294 && ($hour <= 23)
295 && ($min <= 59)) {
296 return TRUE;
297 }
298 }
299 return FALSE;
300}

329function ComputeArgCounts($commands, $argv, $cmd, &$actual, &$expected, &$acceptable, &$check)
330{
331 $actual= count($argv);
332 $expected= count($commands[$cmd]['argv']);
333
334 $acceptable= $expected;
335 for ($argpos= 0; $argpos < $expected; $argpos++) {
336 $argtype= $commands[$cmd]['argv'][$argpos];
337 if ($argtype & NONE) {
338 $acceptable--;
339 }
340 }
341
343 $check= min($actual, $expected);

359function ValidateArgs($commands, $command, $argv, $check)
360{
361 global $ArgTypes;
362
363 $helpmsg= $commands[$command]['desc'];
364 $logmsg= $commands[$command]['desc'];
365
366 $valid= FALSE;
367 // Check each argument in order
368 for ($argpos= 0; $argpos < $check; $argpos++) {
369 $arg= $argv[$argpos];
370 $argtype= $commands[$command]['argv'][$argpos];
371
372 // Multiple types may match for an arg, hence the foreach loop
373 foreach ($ArgTypes as $type => $conf) {
374 // Acceptable types are bitwise ORed, hence the AND here
375 if ($argtype & $type) {
376 $validatefunc= $conf['func'];
377 if ($validatefunc($arg)) {
378 $valid= TRUE;
379
380 if ($type & FILEPATH) {
381 // Further check if file really exists
382 exec("[ -e $arg ]", $output, $retval);
383 if ($retval !== 0) {
384 $valid= FALSE;
385
386 $errormsg= "$command: $arg";
387 Error(_('No such file').": $errormsg");
388 ctlr_syslog(LOG_NOTICE, __FILE__, __FUNCTION__, __LINE__, "No such file: $errormsg");
389 }
390 }
391
392 if ($valid) {
393 // One type succeded, hence do not check for other possible types for this arg
394 break;
395 }
396 }
397 else {
398 $valid= FALSE;
399
400 $helpmsg.= "\n"._($conf['desc']).': '.$arg;
401 $logmsg.= "\n".$conf['desc'].': '.$arg;
402 // Will keep checking if further types are possible for this arg
403 }
404 }
405 }
406
407 if (!$valid) {
408 // One arg failed to check, do not run the func
409 break;
410 }
411 }
412
413 if (!$valid) {
414 Error(_('Arg type check failed').": $helpmsg");
415 ctlr_syslog(LOG_NOTICE, __FILE__, __FUNCTION__, __LINE__, "Arg type check failed: $logmsg");
416 }
417 return $valid;
418}

429function ExpandArgs(&$ArgV, &$Locale, &$View, &$Command)
430{
431 $valid= FALSE;
432
433 // Arg 0 contains all of the args as a json encoded string
434 if (count($ArgV) < 1) {
435 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, 'Not enough args: '.print_r($ArgV, TRUE));
436 goto out;
437 }
438
439 if (count($ArgV) > 1) {
440 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, 'Too many args: '.print_r($ArgV, TRUE));
441 goto out;
442 }
443
444 $decoded= json_decode($ArgV[0], TRUE);
445
446 if ($decoded == NULL || !is_array($decoded)) {
447 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, 'Failed decoding args: '.print_r($ArgV, TRUE));
448 goto out;
449 }
450
451 $ArgV= $decoded;
452
453 if (count($ArgV) < 3) {
454 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, 'Not enough args: '.print_r($ArgV, TRUE));
455 goto out;
456 }
457
458 // Controller runs using the session locale of View
459 $Locale= $ArgV[0];
460 $View= $ArgV[1];
461 $Command= $ArgV[2];
462 // Discard locale, view, and command before computing arg counts
463 $ArgV= array_slice($ArgV, 3);
464
465 $valid= TRUE;
466out:
467 return $valid;
468}

481function ValidateCommand($ArgV, $Locale, $View, $Command, $validateargs, &$Model)
482{
483 global $Models, $LOCALES, $VIEW_PATH;
484
485 $valid= FALSE;
486
487 if (!class_exists($Models[$View])) {
488 Error(_('View not in Models').": $View");
489 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, "View not in Models: $View");
490 goto out;
491 }
492
493 $Model= new $Models[$View]();
494
498 if (!array_key_exists($Locale, $LOCALES)) {
499 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, "Locale not in LOCALES: $Locale");
500 goto out;
501 }
502
503 putenv('LC_ALL='.$Locale);
504 putenv('LANG='.$Locale);
505
506 $Domain= 'pffw';
507 bindtextdomain($Domain, $VIEW_PATH.'/locale');
508 bind_textdomain_codeset($Domain, $LOCALES[$Locale]['Codeset']);
509 textdomain($Domain);
510
511 if (!method_exists($Model, $Command)) {
512 $ErrorStr= "$Models[$View]->$Command()";
513 Error(_('Method does not exist').": $ErrorStr");
514 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, "Method does not exist: $ErrorStr");
515 goto out;
516 }
517
518 if (!array_key_exists($Command, $Model->Commands)) {
519 Error(_('Unsupported command').": $Command");
520 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, "Unsupported command: $Command");
521 goto out;
522 }
523
524 if (!$validateargs) {
525 goto out2;
526 }
527
528 ComputeArgCounts($Model->Commands, $ArgV, $Command, $ActualArgC, $ExpectedArgC, $AcceptableArgC, $ArgCheckC);
529
530 if ($ActualArgC < $AcceptableArgC) {
531 $ErrorStr= "[$AcceptableArgC]: $ActualArgC";
532 Error(_('Not enough args')." $ErrorStr");
533 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, "Not enough args $ErrorStr");
534 goto out;
535 }
536
537 if ($ActualArgC > $ExpectedArgC) {
538 $ErrorStr= "[$ExpectedArgC]: $ActualArgC: ".implode(', ', array_slice($ArgV, $ExpectedArgC));
539 Error(_('Too many args')." $ErrorStr");
540 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, "Too many args $ErrorStr");
541 goto out;
542 }
543
544 // Check only the relevant args
545 if ($ArgCheckC > 0 && !ValidateArgs($Model->Commands, $Command, $ArgV, $ArgCheckC)) {
546 Error(_('Not running command').": $Command");
547 ctlr_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, "Not running command: $Command");
548 goto out;
549 }
550out2:
551 $valid= TRUE;
552out:
553 return $valid;
554}
