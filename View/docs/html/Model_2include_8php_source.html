194function ctlr_syslog($prio, $file, $func, $line, $msg)
195{
196 global $LOG_LEVEL, $LOG_PRIOS;
197
198 $msg= trim($msg);
199 try {
200 openlog('ctlr', LOG_PID, LOG_LOCAL0);
201
202 if ($prio <= $LOG_LEVEL) {
203 $func= $func == '' ? 'NA' : $func;
204 $log= "$LOG_PRIOS[$prio] $file: $func ($line): $msg";
205 if (!syslog($prio, $log)) {
206 $log.= "\n";
207 if (!fwrite(STDERR, $log)) {
208 echo $log;
209 }
210 }
211 }
212 closelog();
213 }
214 catch (Exception $e) {
215 echo 'Caught exception: ', $e->getMessage(), "\n";
216 echo "ctlr_syslog() failed: $prio, $file, $func, $line, $msg\n";
217 // No need to closelog(), it is optional
218 }
219}
