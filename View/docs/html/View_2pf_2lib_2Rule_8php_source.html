579 function inputAddValue($key, $value, $parent= NULL)
580 {
581 $rule= &$this->rule;
582 if ($parent !== NULL) {
583 $rule= &$this->rule[$parent];
584 }
585
586 if (!isset($rule[$key])) {
587 $rule[$key]= $value;
588 } else {
589 if (!is_array($rule[$key])) {
590 // Make array
591 $tmp= $rule[$key];
592 unset($rule[$key]);
593 $rule[$key][]= $tmp;
594 }
595 $rule[$key][]= $value;
596 $rule[$key]= array_unique($rule[$key]);
597 }

664 function inputDelEmptyRecursive($array, $flatten)
665 {
666 foreach ($array as $key => $value) {
667 if ($value == '') {
668 unset($array[$key]);
669 } elseif (is_array($value)) {
672 $array[$key]= $this->inputDelEmptyRecursive($value, in_array($key, array('timeout', 'limit', 'log')) ? FALSE : $flatten);
673
674 if (count($array[$key]) == 0) {
675 // Array is empty, delete it
676 unset($array[$key]);
677 } elseif (count($array[$key]) == 1 && $flatten && !in_array($key, array('timeout', 'limit', 'log'))) {
678 // Array has only one element, convert from array to simple NVP
679 $v= current($array[$key]);
680 unset($array[$key]);
681 $array[$key]= $v;
682 }
683 }
684 }
685 return $array;
686 }
