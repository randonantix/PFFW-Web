32/*
33 * This class provides a filesystem cache. Cache entries can be requested and
34 * are dated at the time of the request.
35 *
36 * The current implementation gets the date once at class creation. All files
37 * in the cache are evaluated on class creation and are deleted if eligable.
38 *
39 * $symon['cache_keepstale'] : set to disable file deletion.
40 *
41 * Typical use:
42 *
43 * $key = $cache->insert(content, <content_identifiers>+);
44 * $filename = $cache->getfilename($key);
45 * $cache->expire($key);
46 *
47 * Use obtain_file_cache(key) to get a direct filename to a cache entry with
48 * key <key>.
49 */
50require_once("setup.inc");
51require_once("tools.inc");
52
53class Cache {
54 var $db;
55 var $timestamp;
56
57 function __construct() {
58 global $symon;
59
60 if (!isset($symon["cache_duration"])) {
61 config_error("cache_duration", "is not set");
62 return 0;
63 }
64 if (!isset($symon["cache_dir"])) {
65 config_error("cache_dir", "is not set");
66 return 0;
67 }
68 if (!is_dir($symon["cache_dir"])) {
69 config_error("cache_dir", "does not point to a directory");
70 return 0;
71 }
72
73 return $this->_load();
74 }
75
76 function _load() {
77 global $symon;
78
79 $ts = time();
80 $this->timestamp = $ts - ($ts % $symon["cache_duration"]);
81
82 $dir = dir($symon["cache_dir"]);
83 while ($file = $dir->read()) {
84 if ($file == '.' || $file == '..') {
85 continue;
86 }
87 if (!is_file($symon["cache_dir"].'/'.$file)) {
88 runtime_error("'".$file."' is not a cache file");
89 return 0;
90 }
91
92 $this->_cache($file);
93 }
94 }
95
96 function _cache($filename) {
97 global $symon;
98
99 if (preg_match("/^([0-9]+)_([0-9a-f]+)(\.[a-z]+)?$/i",
100 $filename, $match)) {
101 $this->db[$match[2]] = $filename;
102 if (($this->timestamp > $match[1]) &&
103 (!isset($symon['cache_keepstale']))) {
104 $this->expire_key($match[2]);
105 }
106 } else {
107 runtime_error("'".$filename."' is not a cache file");
108 return 0;
109 }
110 }
111
112 function _iscached($filename) {
113 if (preg_match("/^([0-9]+)_([0-9a-f]+)(\.[a-z]+)?$/i",
114 $filename, $match)) {
115 return isset($this->db[$match[2]]);
116 } else {
117 if (preg_match("/^([0-9a-f]+)(\.[a-z]+)?$/i",
118 $filename, $match)) {
119 return isset($this->db[$match[1]]);
120 } else {
121 return isset($this->db[$filename]);
122 }
123 }
124 }
125
126 function _get($filename) {
127 if (preg_match("/^([0-9]+)_([0-9a-f]+)(\.[a-z]+)?$/i",
128 $filename, $match)) {
129 if (isset($this->db[$match[2]])) {
130 return $this->db[$match[2]];
131 }
132 } else {
133 if (preg_match("/^([0-9a-f]+)(\.[a-z]+)?$/i",
134 $filename, $match)) {
135 if (isset($this->db[$match[1]])) {
136 return $this->db[$match[1]];
137 }
138 } else {
139 if (isset($this->db[$filename])) {
140 return $this->db[$filename];
141 }
142 }
143 }
144
145 return undef;
146 }

148 function insert(&$content, $extension = '') { /* variadric */
149 $args = '';
150 $maxargs = func_num_args();
151 $arg_list = func_get_args();
152 for ($i = 0; $i < $maxargs; $i++) {
153 if (is_array($arg_list[$i])) {
154 $args .= join($arg_list[$i]);
155 } elseif (is_scalar($arg_list[$i])) {
156 $args .= $arg_list[$i];
157 } else {
158 runtime_error('cache: wrong type of argument '.gettype($arg_list[$i]) .' in "insert"');
159 }
160 }
161
162 $key = md5($args);
163
164 if (!$this->_iscached($key)) {
165 $this->_cache($this->timestamp . '_' . $key . $extension);
166 $filename = $this->getfilename($key);
167 if (strlen($content) > 0) {
168 save($filename, $content);
169 }
170 }
171
172 return $key;
173 }

175 function obtain_filecache($key) {
176 if (!$this->_iscached($key)) {
177 $this->_cache($this->timestamp . '_' . $key);
178 }
179
180 return $this->getfilename($key);
181 }

183 function getfilename($key) {
184 global $symon;
185
186 if ($this->_iscached($key)) {
187 $filename = $this->_get($key);
188 return $symon["cache_dir"].'/'.$filename;
189 } else {
190 return '';
191 }
192 }

194 function expire_key($key) {
195 global $runtime_args;
196
197 $filename = $this->getfilename($key);
198 if (is_file($filename)) {
199 if (!unlink($filename)) {
200 runtime_error("'".$filename."' cannot be deleted");
201 }
202 }
203
204 unset($this->db[$key]);
205 }
206
207 function _expire_all() {
208 if (is_array($this->db)) {
209 foreach($this->db as $key => $value) {
210 $this->expire_key($key);
211 }
212 }
213 }
214
215 function _display() {
216 if (is_array($this->db)) {
217 foreach($this->db as $key => $value) {
218 print "\xa ".$key.'='.$value;
219 }
220 }
221 }
