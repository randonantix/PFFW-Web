32 function dispOption()
33 {
34 ?>
35 <td title="<?php echo _TITLE('Option') ?>" colspan="12">
36 <?php
37 if (isset($this->rule['type'])) {
38 $value= isset($this->rule[$this->rule['type']]) ? $this->rule[$this->rule['type']] : '';
39 if (in_array($this->rule['type'], array('loginterface', 'optimization', 'ruleset-optimization', 'block-policy', 'state-policy', 'debug', 'fingerprints', 'hostid'))) {
40 echo $this->rule['type'] . ": $value";
41 } elseif ($this->rule['type'] == 'skip') {
42 if (!is_array($value)) {
43 echo "skip on $value";
44 } else {
45 echo 'skip on ' . implode(', ', $value);
46 }
47 } elseif ($this->rule['type'] == 'reassemble') {
48 echo $this->rule['type'] . ": $value";
49 if (isset($this->rule['no-df'])) {
50 echo ' no-df';
51 }
52 } elseif ($this->rule['type'] == 'syncookies') {
53 echo $this->rule['type'] . ": $value";
54 if ($value === 'adaptive') {
55 echo ' (start ' . (isset($this->rule['start']) ? $this->rule['start'] : '') . ', end ' . (isset($this->rule['end']) ? $this->rule['end'] : '') . ')';
56 }
57 }
58 }
59 ?>
60 </td>
61 <?php
62 }
