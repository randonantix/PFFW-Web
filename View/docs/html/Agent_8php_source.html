456 $temp = fread($this->fsock, 4);

458 user_error('Connection closed while requesting identities');
459 return array();
460 }

465 user_error('Connection closed while requesting identities');
466 return array();
467 }

576 function _forward_data($data)
577 {
578 if ($this->expected_bytes > 0) {
579 $this->socket_buffer.= $data;
580 $this->expected_bytes -= strlen($data);
581 } else {
582 $agent_data_bytes = current(unpack('N', $data));
583 $current_data_bytes = strlen($data);
584 $this->socket_buffer = $data;
585 if ($current_data_bytes != $agent_data_bytes + 4) {
586 $this->expected_bytes = ($agent_data_bytes + 4) - $current_data_bytes;
587 return false;
588 }
589 }
590
591 if (strlen($this->socket_buffer) != fwrite($this->fsock, $this->socket_buffer)) {
592 user_error('Connection closed attempting to forward data to SSH agent');
593 return false;
594 }
595
596 $this->socket_buffer = '';
597 $this->expected_bytes = 0;
598
599 $temp = fread($this->fsock, 4);
600 if (strlen($temp) != 4) {
601 user_error('Connection closed while reading data response');
602 return false;
603 }
604 $agent_reply_bytes = current(unpack('N', $temp));
605
606 $agent_reply_data = fread($this->fsock, $agent_reply_bytes);
607 if (strlen($agent_reply_data) != $agent_reply_bytes) {
608 user_error('Connection closed while reading data response');
609 return false;
610 }
611 $agent_reply_data = current(unpack('a*', $agent_reply_data));
612
613 return pack('Na*', $agent_reply_bytes, $agent_reply_data);
