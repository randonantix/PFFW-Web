145function GetLogFile()

517function PrintNVPs($nvps, $title, $maxcount= 100, $pie=TRUE, $needle='', $prefix='', $postfix='', $dateArray=array(), $logFile= '')
518{
519 global $IMG_PATH, $View, $FormIdCount;
520 ?>
521 <strong><?php echo $title ?></strong>
522 <table id="stats">
523 <?php
524 arsort($nvps);
525
526 $count= 0;
527 foreach ($nvps as $n => $value) {
528 ?>
529 <tr>
530 <td class="value">
531 <?php echo $value ?>
532 </td>
533 <td class="name">
534 <?php
535 // Empty strings print default gettext header lines otherwise
536 $name= $n !== '' ? _(htmlspecialchars($n)):'-';
537
538 // Needle can be used to disable log searches
539 if ($needle === FALSE) {
540 echo $name;
541 } else {
542 // Need unique ids for each form for submission to work
543 $formId = 'form'.$FormIdCount++;
544
545 // Need SearchRegExp and action URL supplied externally
546 // Regex should search for a case sensitive exact match, otherwise github matches live.github or GITHUB too
547
548 // Caveats of this regexp method:
549 // @todo P3scan Number of e-mails Clean Exit: Searches all, need a separate parser, e.g. for (Clean Exit). Mails: 1
550 // @todo P3scan pop3s: Cannot find pop3s, need to make POP3S lowercase
551 // @todo Smtp-gated Source IPs: Searches and finds all IPs, not just source (or destination), the same issue with other modules
552 // @todo Openssh briefstats password: Searches and finds all password words, not just failed attempts
553 // @todo syslog "last message repeated x times": Causes more date lines in brief stats than can be parsed, need -rr option in OpenBSD 6.x/syslog
554 // @todo Briefstats should use the Total Needle too, otherwise we cannot get general statistics for, say, spamassassin
555
556 // Use default pre/postfixes if not supplied by the caller
557 if ($prefix == '') {
558 // @attention Use [[:blank:]]+, [:blank:]+ does not work
559 $prefix= '([[:blank:]]+)';
560 }
561
562 if ($postfix == '') {
563 $postfix= '([^[:alnum:]]+)';
564 }
565
567 // ~30 secs if search name is long, as in SSLproxy error or warning messages. Two cascaded greps are very fast.
568 // Otherwise, the following would/could do the same job.
569 //$regexp= $needle == '' ?
570 // "($prefix|^)${name}($postfix|$)" :
571 // "(($prefix|^)${name}$postfix.*($needle)|($needle).*$prefix${name}($postfix|$))";
572
573 $regexp= "($prefix|^)".Escape($name, '()[]+?')."($postfix|$)";
574
576 ?>
577 <form id="<?php echo $formId ?>" name="<?php echo $formId ?>" action="<?php echo $View->LogsPage ?>?submenu=archives" method="post">
578 <input type="hidden" name="SearchRegExp" value="<?php echo $regexp ?>" />
579 <input type="hidden" name="SearchNeedle" value="<?php echo $needle ?>" />
580 <input type="hidden" name="Month" value="<?php echo isset($dateArray['Month']) ? $dateArray['Month'] : '' ?>" />
581 <input type="hidden" name="Day" value="<?php echo isset($dateArray['Day']) ? $dateArray['Day'] : '' ?>" />
582 <input type="hidden" name="Hour" value="<?php echo isset($dateArray['Hour']) ? $dateArray['Hour'] : '' ?>" />
583 <?php
584 if ($logFile != '') {
585 ?>
586 <input type="hidden" name="LogFile" value="<?php echo $logFile ?>" />
587 <?php
588 }
589 ?>
590 <input type="hidden" name="Sender" value="Stats" />
591 </form>
592 <a onclick="document.<?php echo $formId ?>.submit()" style="cursor: pointer;" title="<?php echo _TITLE('Click to search in the logs') ?>"><?php echo $name ?></a>
593 <?php
594 }
595 ?>
596 </td>
597 <?php
598 if ($count == 0 && $pie) {
599 ?>
600 <img id="chart" class="chart-trigger" onclick="generateChart(<?php echo str_replace('"', "'", str_replace("'", "\'", json_encode($nvps))) ?>, <?php echo "'".str_replace("'", "\'", $title)."'" ?>);"
601 src="<?php echo $IMG_PATH.'chart.png' ?>" name="<?php echo $title ?>" alt="<?php echo $title ?>" align="absmiddle" >
602 <?php
603 }
604 ?>
605 </tr>
606 <?php
607 if (++$count >= $maxcount) {
608 break;
609 }
610 }
611 ?>
612 </table>
613 <?php
614}

777 if (is_array($value)) {
778 $data[$name]['value']= ceil($data[$name]['value'] / $divisor);
