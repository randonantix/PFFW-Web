397 function hash($text)
398 {
399 if (!empty($this->key) || is_string($this->key)) {
400 switch ($this->engine) {
401 case CRYPT_HASH_MODE_MHASH:
402 $output = mhash($this->hash, $text, $this->computedKey);
403 break;
404 case CRYPT_HASH_MODE_HASH:
405 $output = hash_hmac($this->hash, $text, $this->computedKey, true);
406 break;
407 case CRYPT_HASH_MODE_INTERNAL:
408 $key = str_pad($this->computedKey, $this->b, chr(0)); // step 1
409 $temp = $this->ipad ^ $key; // step 2
410 $temp .= $text; // step 3
411 $temp = call_user_func($this->hash, $temp); // step 4
412 $output = $this->opad ^ $key; // step 5
413 $output.= $temp; // step 6
414 $output = call_user_func($this->hash, $output); // step 7
415 }
416 } else {
417 switch ($this->engine) {
418 case CRYPT_HASH_MODE_MHASH:
419 $output = mhash($this->hash, $text);
420 break;
421 case CRYPT_HASH_MODE_HASH:
422 $output = hash($this->hash, $text, true);
423 break;
424 case CRYPT_HASH_MODE_INTERNAL:
425 $output = call_user_func($this->hash, $text);
426 }
427 }
428
429 return substr($output, 0, $this->l);

816 $hash[4]->add($e),

888 function _add()
889 {
890 static $mod;
891 if (!isset($mod)) {
892 $mod = pow(2, 32);
893 }
894
895 $result = 0;
896 $arguments = func_get_args();
897 foreach ($arguments as $argument) {
898 $result+= $argument < 0 ? ($argument & 0x7FFFFFFF) + 0x80000000 : $argument;
899 }
900
901 switch (true) {
902 case is_int($result):
903 // PHP 5.3, per http://php.net/releases/5_3_0.php, introduced "more consistent float rounding"
904 case version_compare(PHP_VERSION, '5.3.0') >= 0 && (php_uname('m') & "\xDF\xDF\xDF") != 'ARM':
905 // PHP_OS & "\xDF\xDF\xDF" == strtoupper(substr(PHP_OS, 0, 3)), but a lot faster
906 case (PHP_OS & "\xDF\xDF\xDF") === 'WIN':
907 return fmod($result, $mod);
908 }
909
910 return (fmod($result, 0x80000000) & 0x7FFFFFFF) |
911 ((fmod(floor($result / 0x80000000), 2) & 1) << 31);
912 }
