119 function _get_chr_and_nl() {
120 if ($this->eof) {
121 runtime_error('internal: trying to get_chr while eof has been signalled');
122 }
123 if (count($this->unget)) {
124 return array_shift($this->unget);
125 }
126 $item = '';
127 while ($item == '') {
128 if (!is_array($this->lineitems) || count($this->lineitems) == 0) {
129 $this->_get_next_line();

131 return;
132 }
133 return "\n";

140 function _get_next_line() {
141 if ($this->eof) {
142 runtime_error('internal: trying to get_next_line while eof has been signalled');
143 }
144 if (count($this->buffer) == 0) {
145 $this->eof = 1;
146 return;
147 }
148 $this->line = array_shift($this->buffer);
149 $this->lineno += 1;
150 $this->lineitems = preg_split('//', $this->line, -1, PREG_SPLIT_NO_EMPTY);
151 }

153 function _next_token() {
154 if ($this->eof) {
155 runtime_error('internal: trying next_token while eof has been signalled');
156 }
157
158 $this->token = '';
159 $ch = $this->_get_chr();
160
161 while ($this->eof || !(strpos(" \t\r\n#", $ch) === FALSE)) {
162 if ($this->eof) {
163 return;
164 }
165 if ($ch == '#') {
166 $this->_get_next_line();
167 }
168 $ch = $this->_get_chr();
169 }
170
171 if ($this->eof) {
172 $this->token = '';
173 return $this->token;
174 }
175
176 if ($ch == '"') {
177 $this->token = $this->_get_quoted_string('"');
178 return $this->token;
179 }
180 if ($ch == "'") {
181 $this->token = $this->_get_quoted_string("'");
182 return $this->token;
183 }
184 if (!(strpos($this->keywordchrs, $ch) === FALSE)) {
185 $this->token = '';
186 while (!(strpos($this->keywordchrs, $ch) === FALSE) || $this->eof) {
187 $this->token .= $ch;
188 $ch = $this->_get_chr();
189 }
190 if (!$this->eof) {
191 $this->_unget_chr($ch);
192 }
193 return $this->token;
194 }
195 $this->token = $ch;
196 return $this->token;
197 }

199 function _get_quoted_string($quote) {
200
201 $start_lineno = $this->lineno;
202 $this->token = '';
203 $delimited = 0;
204
205 $ch = $this->_get_chr_and_nl();
206 while ($ch != $quote) {
207 if ($this->eof) {
208 $this->parse_error("starting $quote without closing $quote", $start_lineno);
209 return;
210 }
211 if ($ch == "\\" && $quote == '"') {
212 $ch = $this->_get_chr_and_nl();
213 }
214 $this->token .= $ch;
215 $ch = $this->_get_chr_and_nl();
216 }
217 return $this->token;
218 }
