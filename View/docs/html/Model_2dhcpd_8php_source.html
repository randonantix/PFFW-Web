142 function AddIf($if)
143 {

145 $this->DelIf($if);
146 if (($ifs= $this->_getIfs()) !== FALSE) {
147 $ifs= explode("\n", $ifs);

160 function DelIf($if)
161 {
162 if (($ifs= $this->_getIfs()) !== FALSE) {
163 $ifs= explode("\n", $ifs);
164
165 // Don't use: unset($ifs[array_search($if, $ifs)]), check strictly
166 if (($key= array_search($if, $ifs)) !== FALSE) {
167 unset($ifs[$key]);
168 $ifs= implode(' ', $ifs);
169 return $this->SetIfs($ifs);
170 }
171 }
172 return FALSE;
173 }

363 function _getLeases($re= '', $end= 0, $count= 0)
364 {
365 global $Re_Ip;
366
367 //lease 192.168.9.3 {
368 // starts 1 2017/10/30 16:32:46 UTC;
369 // ends 2 2017/10/31 04:32:46 UTC;
370 // hardware ethernet 08:62:66:b9:b4:c5;
371 // client-hostname "soner";
372 //}
373 //lease 192.168.9.2 {
374 // starts 1 2017/10/30 15:03:45 UTC;

381 $re_starts= '\s*starts\s+(\d+)\s+(\d+\/\d+\/\d+)\s+(\d+:\d+:\d+\s+\w*)\s*';
382 $re_ends= '\s*ends\s+(\d+)\s+(\d+\/\d+\/\d+)\s+(\d+:\d+:\d+\s+\w*)\s*';
383 $re_mac= '\s*hardware\s+\w+\s+(\w+:\w+:\w+:\w+:\w+:\w+)\s*';
384 $re_uid= '\s*uid\s+(.+)\s*';
385 $re_host= '\s*(client-hostname|hostname)\s+"(.+)"\s*';
386 $re_status= '\s*(\S+)\s*';
387
388 $re_lease= "/lease\s+($Re_Ip)\s*\{($re_starts;|)($re_ends;|)($re_mac;|)($re_uid;|)($re_status;|)($re_host;|)\s*\}/m";
389
390 $lines= $this->GetFile($this->leasesFile);
391
392 $logs= array();
393 if (preg_match_all($re_lease, $lines, $match, PREG_SET_ORDER)) {
394 $start= $end - $count;
395 $line= 0;
396 $lineCount= 0;
397 foreach ($match as $fields) {
398 $cols['IP']= $fields[1];
399 $cols['Start']= "$fields[4] $fields[5]";
400 $cols['End']= "$fields[8] $fields[9]";
401 $cols['MAC']= $fields[11];
402 $cols['Status']= isset($fields[15]) ? $fields[15] : '';
403 $cols['Host']= isset($fields[18]) ? $fields[18] : '';
404
406 if (preg_match('/'.Escape($re, '/').'/m', $fields[0]) && (($end == 0 && $count == 0) || ($line >= $start && $lineCount < $count))) {
407 $logs[]= $cols;
408 $lineCount++;
409 if ($count > 0 && $lineCount >= $count) {
410 break;
411 }
