279 function parsePfQueueInfo($lines)
280 {
281 $queues= array();
282 $q= array();
283
284 //queue std on em1 bandwidth 100M qlimit 50
285 // [ pkts: 0 bytes: 0 dropped pkts: 0 bytes: 0 ]
286 // [ qlength: 0/ 50 ]
287 foreach ($lines as $line) {
288 if (preg_match('/^queue\s+(\S+)/', $line, $match)) {
289 if (!isset($q['name'])) {
290 $q['name']= '';
291 }
292 $queues[]= $q;
293
294 $q= array('name' => $match[1]);
295 } elseif (preg_match('/^\s*\[\s*pkts:\s*(\d+)\s*bytes:\s*(\d+)\s*dropped pkts:\s*(\d+)\s*bytes:\s*(\d+)\s*\]/', $line, $match)) {
296 $q['pkts']= convertDecimal($match[1]);
297 $q['bytes']= convertBinary($match[2]);
298 $q['droppedPkts']= convertDecimal($match[3]);
299 $q['droppedBytes']= convertBinary($match[4]);
300 } elseif (preg_match('/^\s*\[\s*qlength:\s*(\d+)\s*\/\s*(\d+)\s*/', $line, $match)) {
301 $q['length']= $match[1] . '/' . $match[2];
302 } else {
303 ctlr_syslog(LOG_WARNING, __FILE__, __FUNCTION__, __LINE__, "Failed parsing queue line: $line");
304 }
305 }
306

308 $queues[]= $q;
309 }
310 return $queues;
