589 function _mdsrem($A, $B)
590 {
591 // No gain by unrolling this loop.
592 for ($i = 0; $i < 8; ++$i) {
593 // Get most significant coefficient.
594 $t = 0xff & ($B >> 24);
595
596 // Shift the others up.
597 $B = ($B << 8) | (0xff & ($A >> 24));
598 $A<<= 8;
599
600 $u = $t << 1;
601
602 // Subtract the modular polynomial on overflow.
603 if ($t & 0x80) {
604 $u^= 0x14d;
605 }
606
607 // Remove t * (a * x^2 + 1).
608 $B ^= $t ^ ($u << 16);
609
610 // Form u = a*t + t/a = t*(a + 1/a).
611 $u^= 0x7fffffff & ($t >> 1);
612
613 // Add the modular polynomial on underflow.
614 if ($t & 0x01) {
615 $u^= 0xa6 ;
616 }
617
618 // Remove t * (a + 1/a) * (x^3 + x).
619 $B^= ($u << 24) | ($u << 8);
620 }
621
622 return array(
623 0xff & $B >> 24,
624 0xff & $B >> 16,
625 0xff & $B >> 8,
626 0xff & $B);
627 }
