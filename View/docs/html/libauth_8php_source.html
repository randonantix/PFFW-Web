67function wui_syslog($prio, $file, $func, $line, $msg)
68{
69 global $LOG_LEVEL, $LOG_PRIOS;
70
71 $msg= trim($msg);
72 try {
73 openlog('wui', LOG_PID, LOG_LOCAL0);
74
75 if ($prio <= $LOG_LEVEL) {
76 $user= array_key_exists('USER', $_SESSION) ? $_SESSION['USER'] : 'NA';
77 $useratip= $user.'@'.filter_input(INPUT_SERVER, 'REMOTE_ADDR');
78 $func= $func == '' ? 'NA' : $func;
79 $log= "$LOG_PRIOS[$prio] $useratip $file: $func ($line): $msg";
80 if (!syslog($prio, $log)) {
81 $log.= "\n";
82 if (!fwrite(STDERR, $log)) {
83 echo $log;
84 }
85 }
86 }
87 closelog();
88 }
89 catch (Exception $e) {
90 echo 'Caught exception: ', $e->getMessage(), "\n";
91 echo "wui_syslog() failed: $prio, $file, $func, $line, $msg\n";
92 // No need to closelog(), it is optional
93 }
94}

133function Authentication($passwd)
134{
135 global $ALL_USERS, $SessionTimeout, $View;
136
137 wui_syslog(LOG_DEBUG, __FILE__, __FUNCTION__, __LINE__, 'Login attempt');
138
139 if (!in_array($_SESSION['USER'], $ALL_USERS)) {
140 wui_syslog(LOG_NOTICE, __FILE__, __FUNCTION__, __LINE__, 'Not a valid user');
141 // Throttle authentication failures
142 exec('/bin/sleep 5');
143 LogUserOut('Authentication failed');
144 }
145
146 if (!$View->CheckAuthentication($_SESSION['USER'], $passwd)) {
147 wui_syslog(LOG_NOTICE, __FILE__, __FUNCTION__, __LINE__, 'Password mismatch');
148 // Throttle authentication failures
149 exec('/bin/sleep 5');
150 LogUserOut('Authentication failed');
151 }
152 wui_syslog(LOG_DEBUG, __FILE__, __FUNCTION__, __LINE__, 'Authentication succeeded');
153
154 // Encrypt the password to save as a cookie var.
155 $random= exec('(dmesg; sysctl; route -n show; df; ifconfig -A; hostname) | cksum -q -a sha256 -');
156
157 $cryptKey= pack('H*', $random);
158
159 $iv_size= openssl_cipher_iv_length('AES-256-CBC');
160 $iv= openssl_random_pseudo_bytes($iv_size);
161
162 $ciphertext= openssl_encrypt($passwd, 'AES-256-CBC', $cryptKey, OPENSSL_RAW_DATA, $iv);
163 if (!$ciphertext) {
164 wui_syslog(LOG_ERR, __FILE__, __FUNCTION__, __LINE__, 'openssl_encrypt failed: '.openssl_error_string());
165 exit;
166 }
167 $ciphertext= $iv.$ciphertext;
168
169 $ciphertext_base64= base64_encode($ciphertext);
170
171 // Save password to the cookie
172 setcookie('passwd', $ciphertext_base64);
173
174 // Save key to the session, so we can decrypt the password from the cookie later on to log in to the SSH server
175 $_SESSION['cryptKey']= $cryptKey;
176
177 // Update session timeout now, otherwise in the worst case scenario, vars.php may log user out on very close session timeout
178 $_SESSION['Timeout']= time() + $SessionTimeout;
179
180 // Reset the pf session if the user changes
181 if (isset($_SESSION['PREVIOUS_USER']) && $_SESSION['PREVIOUS_USER'] !== $_SESSION['USER']) {
183 unset($_SESSION['pf']);
184 }
185
186 header('Location: /system/info.php?submenu=dashboard');
187 exit;
188}
