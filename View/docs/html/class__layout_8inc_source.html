101 function _generate_host($machine) {
102 global $symon;
103
104 if (!is_dir($symon["host_tree"])) {
105 config_error("host_tree", "does not point to a directory");
106 return 0;
107 }
108
109 if (is_file($symon["host_tree"]."/".$machine)) {
110 runtime_error('$symon["host_tree"] should contain machine directories, not files. ($symon["host_tree"]/cpu0.rrd => $symon["host_tree"]/localhost/cpu0.rrd)');
111 }
112
113 if ($machine != '.' &&
114 $machine != '..' &&
115 is_dir($symon["host_tree"]."/".$machine)) {
116 $groupvars = new Vars();
117 $groupvars->set('host', $machine);
118 $groupvars->set('dir', $symon["host_tree"].'/'.$machine);
119 $groupvars->set('name', $machine);
120 $this->_add_group($groupvars);
121 /* get all graphs */
122 $graph_dir = dir($symon["host_tree"].'/'.$machine);
123 $files = array();
124 while ($item = $graph_dir->read()) {
125 if ($item != '.' &&
126 $item != '..' &&
127 preg_match("/.rrd$/", $item)) {
128 $files[] = $item;
129 }
130 }
131 asort($files);
132 /* get combined graphs */
133 foreach ($symon['combine'] as $type => $enabled) {
134 if ($enabled) {
135 $combine = array();
136 $newfiles = array();
137 foreach ($files as $key => $item) {
138 if (preg_match("/^".$type."[0-9_]/", $item)) {
139 $combine[] = $item;
140 } else {
141 $newfiles[] = $item;
142 }
143 }
144
145 $files = $newfiles;
146
147 if (count($combine)) {
148 $v = new Vars();
149 foreach ($combine as $key => $item) {
150 $v->set('rrdfile'.$key, $symon["host_tree"].'/'.$machine.'/'.$item);
151 }
152 $g = new Graph($groupvars);
153 $g->set_graph_vars($v);
154 $this->_add_item($machine, $g);
155 }
156 }
157 }
158
159 foreach ($files as $key => $item) {
160 $v = new Vars();
161 $v->set('rrdfile', $symon["host_tree"].'/'.$machine.'/'.$item);
162 $g = new Graph($groupvars);
163 $g->set_graph_vars($v);
164 $this->_add_item($machine, $g);
165 }
166 }
167 }

169 function _add_group(&$vars) {
170 if (strtolower(get_class($vars)) != 'vars') {
171 print_r($vars);
172 runtime_error('layout: internal: add_group(vars) - vars are not variables');
173 }
174
175 if (!$vars->defp('name')) {
176 $this->_display();
177 runtime_error('layout: need a valid group name for a group to add');
178 }
179
180 $name = $vars->get('name');
181 if (isset($this->groupvars[$name])) {
182 $this->_display();
183 runtime_error("layout: group $name redefined");
184 }
185 $this->groupvars[$name] = $vars;
186 $this->groups[$name] = array();
187 }

189 function _add_item($groupname, &$value) {

203 function _load($name) {
204 global $symon;
205
206 if ($name == '' || $name == 'default' || $name == 'empty') {
207 runtime_error('layout: need a valid layout name to load');
208 return 0;
209 }
210
211 $name = normalise_filename($name);
212 $file = $symon["layout_dir"].'/'.$name.'.layout';
213
214 $lexer = new Lexer();
215 $lexer->load($name);
216 $groupvars = 0;
217
218 $token = $lexer->next_token();
219 do {
220 if ($token === 0) {
221 $token_string = '';
222 } else {
223 $token_string = $token;
224 }
225
226 switch ($token_string) {
227 case 'group':
228 $groupvars = new Vars();
229 $groupvars->parse($lexer);
230 $lexer->parse_semicolon();
231 $this->_add_group($groupvars);
232 $groupname = $groupvars->get('name');
233 break;
234
235 case 'graph':
236 $graph = new Graph($groupvars);
237 $graph->parse($lexer);
238 $lexer->parse_semicolon();
239 $this->_add_item($groupname, $graph);
240 break;
241
242 case 'text':
243 $text = new Text();
244 $text->parse($lexer);
245 $lexer->parse_semicolon();
246 $this->_add_item($groupname, $text);
247 break;
248
249 default:
250 $lexer->parse_error('Expected group|graph|text');
251 break;
252 }
253
254 $token = $lexer->next_token();
255 } while (!$lexer->is_eof());
256
257 if (isset($symon['layout_debug'])) {
258 $this->_display();
259 }
260 }

262 function render($html= true) {
263 $LinkNames = array(
264 _TITLE('CPU Load'),
265 _TITLE('Temperature'),
266 _TITLE('Fan'),
267 _TITLE('Memory Usage'),
268 _TITLE('Disk I/O'),
269 _TITLE('Partitions Usages'),
270 _TITLE('Loopback Interface'),
271 _TITLE('Internal Interface'),
272 _TITLE('External Interface'),
273 _TITLE('State Statistics'),
274 _TITLE('State Searches vs Packets'),
275 _TITLE('Data Transfer'),
276 _TITLE('Mbuf Statistics'),
277 _TITLE('Network Protocol Usage'),
278 _TITLE('Number of Processes'),
279 _TITLE('Process CPU Load'),
280 );
281
282 $graphs= array();
283
284 reset($this->groups);
285 if (is_array($this->groups)) {
286 foreach($this->groups as $group => $items) {
287 if (array_key_exists($group, $LinkNames)) {
288 $LinkName= $LinkNames[$group];
289 }
290 else {
291 $LinkName= $group;
292 }
293
294 if ($html) {
295 print ' <div class="group">
296 <h3 class="header" style="margin: 4px 0 2px 0;"><a name="' . $group . '">' . _($LinkName) . '</a></h3>
297 </div>';
298 }
299
300 reset($items);
301 foreach($items as $item) {
302 $graphs[$LinkName]= $item->render($html);
303 }
304
305 if ($html) {
306 print '<div class="item-footer"></div>';
307 }
308 }
309 }
310 return $graphs;
311 }
