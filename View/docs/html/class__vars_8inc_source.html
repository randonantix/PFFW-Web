112 function parse(&$lexer) {
113 $token = '';

115
116 while (!$done_parsing) {
117 $name = $lexer->next_token();
118
119 $equals = $lexer->next_token();
120 if ($equals != '=') {

122 }

124 $value = $lexer->next_token();

129 if ($token != ',') {
