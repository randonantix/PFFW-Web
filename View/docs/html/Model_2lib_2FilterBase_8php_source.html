345 {

349 } elseif ($this->words[$this->index + 1] === 'tos') {

421 function genSrcDest()
422 {
423 if (isset($this->rule['all'])) {
424 $this->str.= ' all';
425 } else {
426 if (isset($this->rule['from']) || isset($this->rule['fromroute']) || isset($this->rule['fromport'])) {
427 $this->str.= ' from';
428 $this->genItems('from');
429 $this->genValue('fromroute', 'route ');
430 $this->genItems('fromport', 'port');
431 }
432
433 if (isset($this->rule['os'])) {
434 $this->genItems('os', 'os');
435 }
436
437 if (isset($this->rule['to']) || isset($this->rule['toroute']) || isset($this->rule['toport'])) {
438 $this->str.= ' to';
439 $this->genItems('to');
440 $this->genValue('toroute', 'route ');
441 $this->genItems('toport', 'port');
442 }
443 }
