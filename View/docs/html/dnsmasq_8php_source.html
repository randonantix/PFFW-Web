87 function ParseLogLine($logline, &$cols)
88 {
89 if ($this->ParseSyslogLine($logline, $cols)) {
90 // query[A] www.openbsd.org from 192.168.0.2
91 // query[AAAA] 1.ubuntu.pool.ntp.org from 192.168.0.2
92 $re= "/query\[(\S+)\]\s+(\S+)\s+from\s+(\S+)/";
93 if (preg_match($re, $cols['Log'], $match)) {
94 // Type field is for statistics only, not shown on Logs pages
95 $cols['Type']= $match[1];
96 $cols['Domain']= $match[2];
97 $cols['IP']= $match[3];
98 $cols['Reason']= 'query';
99 // Log field is displayed on the Log column on Logs pages
100 // Since we have further parsed the Log field now, update it
101 $cols['Log']= $cols['Reason'].' '.$cols['Type'];
102 } else {
103 // cached yahoo.com is 74.6.231.20
104 // cached detectportal.firefox.com is <CNAME>
105 // cached connectivity-check.ubuntu.com is NODATA-IPv6
106 // cached 192.168.190.1 is NXDOMAIN
107 if (preg_match("/^cached\s+(\S+)\s+is\s+/", $cols['Log'])) {
108 $cols['Reason']= 'cached';
109 $cols['Log']= htmlentities($cols['Log']);
110 } else {
111 // config error is REFUSED
112 if (preg_match('/\s+is\s+REFUSED$/', $cols['Log'])) {
113 $cols['Reason']= 'REFUSED';
114 }
115 }
116 }
117 return TRUE;
118 }
119 return FALSE;
120 }
